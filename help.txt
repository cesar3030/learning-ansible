# executer une cmd shell
ansible all -m shell -a 'echo $TERM'

# copier un fichier dans tous les serveurs
ansible all -m copy -a "src=/tmp/test.txt dest=./"

# change les permissions d'un fichier (on peut aussi changer le owner, le group)
ansible all -m shell -a 'ls -l'
ansible all -m file -a "dest=~/test.txt mode=777"

# cree un dossier
ansible all -m file -a "dest=~/dir_from/ansible mode=755 owner=cesar group=cesar state=directory"

# delete un dossier recussivement 
ansible all -m file -a "dest=~/dir_from state=absent"

# check qu'un packet est installe
ansible all -m apt -a "name=acme state=present"
